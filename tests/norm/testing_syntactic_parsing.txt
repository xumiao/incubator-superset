Sentence(words: [String], np: NP, vp: VP) = Split(words)?s & ( NP(words[:s])?np & VP(words[s:])?vp
                                                             | NP(words[:s])?np );
NP(words: [String], article: Article, adjective: Adj, noun: Norn) =
      Article(words[0])?article & Split(words[1:])?s & Adj(words[1:(s+1)]) & Noun(words[(s+1):])
    | Split(words)?s & Adj(words[:s]) & Noun(words[s:]);

VP(words: [String], verb: Verb, np: NP) = Split(words)?s & Verb(words[:s])?verb & NP(words[s:])?np;

Article(word: String) = 'the' | 'a';

Verb(words: [String]) = ['see'] | ['go' 'to'];

Noun(words: [String]) = ['New' 'York'] | ['I'];

Adj(words: [String]) = ['beautiful'];

Split(words: [String]): Integer = range(1, words.count)?output;
